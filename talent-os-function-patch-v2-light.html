functionSelect.addEventListener("change", e => {
  const fn = FUNCTIONS.find(f => f.id === e.target.value);
  if (fn && fn.subs) {
    subFnLabel.style.display = "";
    subFunctionSelect.innerHTML = fn.subs.map(s =>

    subFunctionSelect.required = false;

  }
});
3 · Roles page — tab bar + sub-bar wiring 
let currentFunction = "all";
let currentSub = null;

document.querySelectorAll(".fn-tab").forEach(tab => {
  tab.addEventListener("click", () => {
    document.querySelectorAll(".fn-tab").forEach(t => t.setAttribute("aria-selected","false"));
    tab.setAttribute("aria-selected","true");
    currentFunction = tab.dataset.fn;
    currentSub = null;

    // Show / hide the right sub-bar
    document.querySelectorAll(".sub-bar").forEach(b => b.classList.remove("is-visible"));
    const subBar = document.getElementById("sub-" + currentFunction);
    if (subBar) subBar.classList.add("is-visible");

    renderRoles();
  });
});

document.querySelectorAll(".sub-chip").forEach(chip => {
  chip.addEventListener("click", () => {
    const parent = chip.parentElement;
    parent.querySelectorAll(".sub-chip").forEach(c => c.setAttribute("aria-selected","false"));
    chip.setAttribute("aria-selected","true");
    currentSub = chip.textContent.trim().split(" ")[0] === "All" ? null : chip.textContent.trim().split(" ")[0];
    renderRoles();
  });
});

function visibleRoles() {
  return roles.filter(r => {
    if (currentFunction !== "all" && r.function !== currentFunction) return false;
    if (currentSub && r.subFunction !== currentSub) return false;
    return true;
  });
}

// Aggregation
function dashByFunction() {
  return FUNCTIONS.map(f => {
    const inFn = roles.filter(r => r.function === f.id);
    const out = {
      ...f,
      open:  inFn.filter(r => r.status === "Open").length,
      hired: inFn.filter(r => r.status === "Hired").length,
      hold:  inFn.filter(r => r.status === "On hold").length,
    };
    if (f.subs) {
      out.subs = f.subs.map(s => {
        const inSub = inFn.filter(r => r.subFunction === s);
        return {
          name: s,
          open:  inSub.filter(r => r.status === "Open").length,
          hired: inSub.filter(r => r.status === "Hired").length,
          hold:  inSub.filter(r => r.status === "On hold").length,
        };
      });
    }
    return out;
  });
}

function applyFilters(roles) {
  return roles
    .filter(r => period.contains(r.openedAt))
    .filter(r => fn   === "all" || r.function    === fn)
    .filter(r => sub  === "all" || r.subFunction === sub)
    .filter(r => dept === "all" || r.department  === dept);
}

function ttHByFunction(roles) {
  return FUNCTIONS.map(f => {
    const closed = roles.filter(r => r.function === f.id && r.status === "Hired");
    const avg = closed.length
      ? closed.reduce((sum, r) => sum + r.daysToHire, 0) / closed.length
      : null;
    return { label: f.label, days: avg };
  }).filter(x => x.days !== null);
}

function inferFunction(role) {
  const t = (role.title + " " + (role.location || "")).toLowerCase();
  if (/warehouse|fulfilment|fulfillment|dwc|jebel ali warehouse/.test(t)) return { fn: "warehouse", sub: null };
  if (/optical lab|lab technician|edger|fitter/.test(t)) return { fn: "lab", sub: null };
  if (/customer service|cs agent|cx support|wa agent/.test(t)) return { fn: "customer_service", sub: null };

  const loc = (role.location || "").toLowerCase();
  if (role.department === "Retail" || /store|optician|optometrist/.test(t)) {
    if (/riyadh/.test(loc))               return { fn: "retail_ksa", sub: "Central" };
    if (/dammam|khobar|jubail/.test(loc)) return { fn: "retail_ksa", sub: "Eastern" };
    if (/jeddah|makkah|medina/.test(loc)) return { fn: "retail_ksa", sub: "Western" };
    if (/dubai|abu dhabi|sharjah|ajman/.test(loc)) return { fn: "retail_gcc", sub: "UAE" };
    if (/kuwait/.test(loc))   return { fn: "retail_gcc", sub: "Kuwait" };
    if (/bahrain|manama/.test(loc)) return { fn: "retail_gcc", sub: "Bahrain" };
    if (/qatar|doha/.test(loc))     return { fn: "retail_gcc", sub: "Qatar" };
    if (/oman|muscat/.test(loc))    return { fn: "retail_gcc", sub: "Oman" };
  }
  return { fn: "hq", sub: null };
}
